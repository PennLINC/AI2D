<script>
document.addEventListener("DOMContentLoaded", function() {
  var headingCells = document.querySelectorAll("th");
  var diagnosesHeading = null;
  for (var i = 0; i < headingCells.length; i++) {
    var headingText = headingCells[i].textContent.replace(/\s+/g, " ").trim();
    if (headingText === "Diagnoses") {
      diagnosesHeading = headingCells[i];
      break;
    }
  }
  if (!diagnosesHeading) return;

  var diagnosesRow = diagnosesHeading.parentElement;
  if (!diagnosesRow || diagnosesRow.children.length < 2) return;
  var diagnosesCell = diagnosesRow.children[1];
  if (!diagnosesCell) return;

  var diagnosisTables = diagnosesCell.querySelectorAll("table");
  for (var j = 0; j < diagnosisTables.length; j++) {
    var diagnosisTable = diagnosisTables[j];
    if (diagnosisTable.getAttribute("data-blsa-toggle-init") === "true") continue;
    diagnosisTable.setAttribute("data-blsa-toggle-init", "true");

    var blockDiv = diagnosisTable.parentElement;
    if (!blockDiv) continue;

    var labelDiv = blockDiv.firstElementChild;
    var label = "diagnosis";
    if (labelDiv && labelDiv.tagName.toLowerCase() === "div") {
      label = labelDiv.textContent.trim();
    }

    diagnosisTable.style.display = "none";

    var button = document.createElement("button");
    button.type = "button";
    button.className = "blsa-dx-toggle-btn";
    button.textContent = "Show " + label + " table";

    (function(table, toggleButton, tableLabel) {
      toggleButton.addEventListener("click", function() {
        var isHidden = table.style.display === "none";
        table.style.display = isHidden ? "table" : "none";
        toggleButton.textContent = (isHidden ? "Hide " : "Show ") + tableLabel + " table";
      });
    })(diagnosisTable, button, label);

    if (labelDiv && labelDiv.tagName.toLowerCase() === "div") {
      labelDiv.insertAdjacentElement("afterend", button);
    } else {
      blockDiv.insertBefore(button, diagnosisTable);
    }
  }
});
</script>
